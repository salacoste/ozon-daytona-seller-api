// Quick script to check actual method counts
import { OzonSellerApiClient } from '../../src/core/client.js';
import { createApiKey, createClientId } from '../../src/core/types.js';

const client = new OzonSellerApiClient({
  apiKey: createApiKey('test-api-key-1234567890123456789012345678901234567890'),
  clientId: createClientId('123456')
});

const apis = [
  ['report', client.report],
  ['premium', client.premium], 
  ['pricesStocks', client.pricesStocks],
  ['betaMethod', client.betaMethod],
  ['promos', client.promos],
  ['pass', client.pass],
  ['cancellation', client.cancellation],
  ['category', client.category],
  ['digital', client.digital],
  ['barcode', client.barcode],
  ['polygon', client.polygon],
  ['sellerRating', client.sellerRating]
];

let total = 0;
apis.forEach(([name, api]) => {
  const methodCount = Object.getOwnPropertyNames(Object.getPrototypeOf(api))
    .filter(name => name !== 'constructor').length;
  console.log(`${name}: ${methodCount}`);
  total += methodCount;
});

console.log(`Total: ${total}`);
