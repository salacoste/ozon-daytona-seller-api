🚀 Быстрый старт

Установка

npm install daytona-ozon-seller-api

Инициализация

import { OzonSellerAPI } from 'daytona-ozon-seller-api'; const api = new OzonSellerAPI({ clientId: 'your-client-id', apiKey: 'your-api-key' });

Первый запрос

// Получить список товаров const products = await api.product.getList({ limit: 10 }); console.log(`Найдено товаров: ${products.result?.items?.length || 0}`); // Получить информацию о заказах FBS const orders = await api.fbs.getOrdersList({ filter: { status: 'awaiting_packaging' }, limit: 20 }); console.log(`Заказов к обработке: ${orders.result?.length || 0}`);

🎯 Примеры использования

Пример: Полный цикл обработки заказа

async function processOrder(orderNumber: string) { // 1. Получаем информацию о заказе const order = await api.fbs.getOrder({ posting_number: orderNumber }); // 2. Проверяем наличие товаров на складе for (const product of order.products) { const stock = await api.product.getStocks({ product_id: product.product_id }); if (stock.present < product.quantity) { throw new Error(`Недостаточно товара ${product.name}`); } } // 3. Упаковываем заказ await api.fbs.packOrder({ posting_number: orderNumber, packages: [{ products: order.products }] }); // 4. Отгружаем заказ await api.fbs.shipOrder({ posting_number: orderNumber, tracking_number: `TRACK${Date.now()}` }); console.log(`✅ Заказ ${orderNumber} успешно обработан`); }
