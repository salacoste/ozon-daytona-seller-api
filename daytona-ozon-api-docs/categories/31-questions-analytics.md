```typescript

  /** –ü–µ—Ä–∏–æ–¥ –∞–Ω–∞–ª–∏–∑–∞ –≤ –¥–Ω—è—Ö */
  analysisPeriodDays: number;
  
  /** –†–∞–∑–º–µ—Ä –≤—ã–±–æ—Ä–∫–∏ –¥–ª—è –∞–Ω–∞–ª–∏–∑–∞ */

interface ProductAnalysis {
  sku: number;
  name: string;
  issues: string[];
  recommendations: string[];
  priority: 'high' | 'medium' | 'low';
  questionsCount: number;
  responseRate: number;
}

class QuestionsAnalyzer {
  private qaApi: QuestionsAnswersApi;
  private config: AnalyticsConfig;

  constructor(qaApi: QuestionsAnswersApi, config: AnalyticsConfig) {
    this.qaApi = qaApi;
    this.config = config;
  }

  /**
   * –ö–æ–º–ø–ª–µ–∫—Å–Ω—ã–π –∞–Ω–∞–ª–∏–∑ —Ç–æ–≤–∞—Ä–æ–≤ —Å –≤–æ–ø—Ä–æ—Å–∞–º–∏
   */
  async performComprehensiveAnalysis(): Promise<ProductAnalysis[]> {
    try {
      // –ü–æ–ª—É—á–µ–Ω–∏–µ –¥–∞–Ω–Ω—ã—Ö –∑–∞ –ø–µ—Ä–∏–æ–¥
      const endDate = new Date();
      const startDate = new Date();
      startDate.setDate(endDate.getDate() - this.config.analysisPeriodDays);

      const analytics = await this.qaApi.getTopSKUWithQuestions({
        limit: this.config.sampleSize,
        filter: {
          date_from: startDate.toISOString(),
          date_to: endDate.toISOString(),
          min_questions_count: this.config.minQuestionsThreshold
        }
      });

      if (analytics.result !== 'success' || !analytics.items) {
        throw new Error('–ù–µ —É–¥–∞–ª–æ—Å—å –ø–æ–ª—É—á–∏—Ç—å –¥–∞–Ω–Ω—ã–µ –∞–Ω–∞–ª–∏—Ç–∏–∫–∏');
      }

      return analytics.items.map(item => this.analyzeProduct(item));
    } catch (error) {
      console.error('‚ùå –û—à–∏–±–∫–∞ –∞–Ω–∞–ª–∏–∑–∞:', error);
      return [];
    }
  }

  /**

  private analyzeProduct(item: any): ProductAnalysis {
    const issues: string[] = [];
    const recommendations: string[] = [];
    let priority: 'high' | 'medium' | 'low' = 'low';

    // –ê–Ω–∞–ª–∏–∑ –ø—Ä–æ—Ü–µ–Ω—Ç–∞ –æ—Ç–≤–µ—Ç–æ–≤
    if (item.answer_rate < this.config.criticalResponseRate) {
      issues.push('–ù–∏–∑–∫–∏–π –ø—Ä–æ—Ü–µ–Ω—Ç –æ—Ç–≤–µ—Ç–æ–≤ –Ω–∞ –≤–æ–ø—Ä–æ—Å—ã');

      priority = 'high';
    } else if (item.answer_rate < 70) {
      issues.push('–°—Ä–µ–¥–Ω–∏–π –ø—Ä–æ—Ü–µ–Ω—Ç –æ—Ç–≤–µ—Ç–æ–≤');
      recommendations.push('–û–ø—Ç–∏–º–∏–∑–∏—Ä–æ–≤–∞—Ç—å –ø—Ä–æ—Ü–µ—Å—Å –æ—Ç–≤–µ—Ç–∞ –Ω–∞ –≤–æ–ø—Ä–æ—Å—ã');
      priority = priority === 'high' ? 'high' : 'medium';
    }

    // –ê–Ω–∞–ª–∏–∑ –∫–æ–ª–∏—á–µ—Å—Ç–≤–∞ –Ω–æ–≤—ã—Ö –≤–æ–ø—Ä–æ—Å–æ–≤
    if (item.new_questions_count > 10) {
      issues.push('–ú–Ω–æ–≥–æ –Ω–µ–æ—Ç–≤–µ—á–µ–Ω–Ω—ã—Ö –≤–æ–ø—Ä–æ—Å–æ–≤');
      recommendations.push('–ü—Ä–∏–æ—Ä–∏—Ç–µ–∑–∏—Ä–æ–≤–∞—Ç—å –æ—Ç–≤–µ—Ç—ã –Ω–∞ –Ω–æ–≤—ã–µ –≤–æ–ø—Ä–æ—Å—ã');
      priority = 'high';
    }

    // –ê–Ω–∞–ª–∏–∑ –æ–±—â–µ–≥–æ –∫–æ–ª–∏—á–µ—Å—Ç–≤–∞ –≤–æ–ø—Ä–æ—Å–æ–≤
    if (item.questions_count > 30) {
      issues.push('–í—ã—Å–æ–∫–∞—è –∞–∫—Ç–∏–≤–Ω–æ—Å—Ç—å –≤–æ–ø—Ä–æ—Å–æ–≤ —É–∫–∞–∑—ã–≤–∞–µ—Ç –Ω–∞ –ø—Ä–æ–±–ª–µ–º—ã –≤ –æ–ø–∏—Å–∞–Ω–∏–∏');
      recommendations.push('–£–ª—É—á—à–∏—Ç—å –æ–ø–∏—Å–∞–Ω–∏–µ —Ç–æ–≤–∞—Ä–∞ –Ω–∞ –æ—Å–Ω–æ–≤–µ —á–∞—Å—Ç—ã—Ö –≤–æ–ø—Ä–æ—Å–æ–≤');
    }

    // –ê–Ω–∞–ª–∏–∑ –æ—Ü–µ–Ω–∫–∏ –æ—Ç–≤–µ—Ç–æ–≤
    if (item.average_answer_rating && item.average_answer_rating < 3.0) {
      issues.push('–ù–∏–∑–∫–∞—è –æ—Ü–µ–Ω–∫–∞ –ø–æ–ª–µ–∑–Ω–æ—Å—Ç–∏ –æ—Ç–≤–µ—Ç–æ–≤');
      recommendations.push('–£–ª—É—á—à–∏—Ç—å –∫–∞—á–µ—Å—Ç–≤–æ –æ—Ç–≤–µ—Ç–æ–≤, –¥–æ–±–∞–≤–∏—Ç—å –±–æ–ª—å—à–µ –¥–µ—Ç–∞–ª–µ–π');
    }

    return {
      sku: item.sku,
      name: item.name,
      issues,
      recommendations,
      priority,
      questionsCount: item.questions_count,
      responseRate: item.answer_rate
    };
  }

  /**
   * –ì–µ–Ω–µ—Ä–∞—Ü–∏—è –æ—Ç—á—ë—Ç–∞ —Å –ø—Ä–∏–æ—Ä–∏—Ç–∏–∑–∞—Ü–∏–µ–π –∑–∞–¥–∞—á

  async generateActionReport(): Promise<string> {

    if (highPriority.length > 0) {
      report += 'üö® –í–´–°–û–ö–ò–ô –ü–†–ò–û–†–ò–¢–ï–¢ (—Ç—Ä–µ–±—É–µ—Ç –Ω–µ–º–µ–¥–ª–µ–Ω–Ω–æ–≥–æ –≤–Ω–∏–º–∞–Ω–∏—è):\n\n';
      highPriority.forEach(analysis => {
```
