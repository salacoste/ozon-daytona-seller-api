## Complex Scenarios

### Delivery Zone Management System

–ö–ª–∞—Å—Å –¥–ª—è –∫–æ–º–ø–ª–µ–∫—Å–Ω–æ–≥–æ —É–ø—Ä–∞–≤–ª–µ–Ω–∏—è –∑–æ–Ω–∞–º–∏ –¥–æ—Å—Ç–∞–≤–∫–∏:

```typescript
class DeliveryZoneManager {
  constructor(private ozonApi: OzonApi) {}
  
  /**
   * –°–æ–∑–¥–∞–Ω–∏–µ –ø–æ–ª–∏–≥–æ–Ω–∞-–¥–æ–Ω—É—Ç–∞ (—Å –æ—Ç–≤–µ—Ä—Å—Ç–∏–µ–º)
   */
  async createDonutPolygon(
    outerBounds: number[][],
    innerBounds: number[][],
    deliveryMethodId: number,
    deliveryTime: number,
    warehouseLat: string,
    warehouseLon: string
  ) {
    try {
      // –°–æ–∑–¥–∞—Ç—å –ø–æ–ª–∏–≥–æ–Ω —Å –æ—Ç–≤–µ—Ä—Å—Ç–∏–µ–º (donut shape)
      const coordinates = `[${JSON.stringify(outerBounds)}, ${JSON.stringify(innerBounds)}]`;
      
      const polygon = await this.ozonApi.polygon.createDeliveryPolygon({
        coordinates
      });
      
      if (!polygon.polygon_id) {
        throw new Error('–ù–µ —É–¥–∞–ª–æ—Å—å —Å–æ–∑–¥–∞—Ç—å –ø–æ–ª–∏–≥–æ–Ω-–¥–æ–Ω—É—Ç');
      }
      
      // –ü—Ä–∏–≤—è–∑–∞—Ç—å –∫ –º–µ—Ç–æ–¥—É –¥–æ—Å—Ç–∞–≤–∫–∏
      await this.ozonApi.polygon.bindPolygonToDeliveryMethod({
        delivery_method_id: deliveryMethodId,
        warehouse_location: {
          lat: warehouseLat,
          lon: warehouseLon
        },
        polygons: [{
          polygon_id: polygon.polygon_id,
          time: deliveryTime
        }]
      });
      
      return polygon.polygon_id;
    } catch (error) {
      console.error('–û—à–∏–±–∫–∞ —Å–æ–∑–¥–∞–Ω–∏—è –ø–æ–ª–∏–≥–æ–Ω–∞-–¥–æ–Ω—É—Ç–∞:', error);
      throw error;
    }
  }
  
  /**
   * –°–æ–∑–¥–∞–Ω–∏–µ —Ä–∞–¥–∏–∞–ª—å–Ω–æ–≥–æ –ø–æ–ª–∏–≥–æ–Ω–∞ (–ø—Ä–∏–±–ª–∏–∂–µ–Ω–∏–µ –∫ –∫—Ä—É–≥—É)
   */
  createCircularPolygon(centerLat: number, centerLon: number, radiusKm: number, points = 12) {
    const coordinates: number[][] = [];
    const earthRadius = 6371; // –∫–º
    
    for (let i = 0; i <= points; i++) {
      const angle = (i * 2 * Math.PI) / points;
      
      // –í—ã—á–∏—Å–ª–µ–Ω–∏–µ –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç —Ç–æ—á–∫–∏ –Ω–∞ –æ–∫—Ä—É–∂–Ω–æ—Å—Ç–∏
      const deltaLat = (radiusKm / earthRadius) * Math.cos(angle) * (180 / Math.PI);
      const deltaLon = (radiusKm / earthRadius) * Math.sin(angle) * (180 / Math.PI) / Math.cos(centerLat * Math.PI / 180);
      
      const lat = centerLat + deltaLat;
      const lon = centerLon + deltaLon;
      
      coordinates.push([lat, lon]);
    }
    
    return `[${JSON.stringify(coordinates)}]`;
  }
  
  /**
   * –°–æ–∑–¥–∞–Ω–∏–µ –∑–æ–Ω—ã –±—ã—Å—Ç—Ä–æ–π –¥–æ—Å—Ç–∞–≤–∫–∏ –≤ —Ä–∞–¥–∏—É—Å–µ –æ—Ç —Å–∫–ª–∞–¥–∞
   */
  async setupExpressDeliveryZone(
    warehouseLat: number,
    warehouseLon: number,
    radiusKm: number,
    deliveryMethodId: number,
    deliveryTimeMinutes: number
  ) {
    try {
      const coordinates = this.createCircularPolygon(warehouseLat, warehouseLon, radiusKm);
      
      const polygon = await this.ozonApi.polygon.createDeliveryPolygon({
        coordinates: `[${coordinates}]`
      });
      
      if (!polygon.polygon_id) {
        throw new Error('–ù–µ —É–¥–∞–ª–æ—Å—å —Å–æ–∑–¥–∞—Ç—å –∑–æ–Ω—É —ç–∫—Å–ø—Ä–µ—Å—Å-–¥–æ—Å—Ç–∞–≤–∫–∏');
      }
      
      await this.ozonApi.polygon.bindPolygonToDeliveryMethod({
        delivery_method_id: deliveryMethodId,
        warehouse_location: {
          lat: warehouseLat.toString(),
          lon: warehouseLon.toString()
        },
        polygons: [{
          polygon_id: polygon.polygon_id,
          time: deliveryTimeMinutes
        }]
      });
      
      console.log(`‚úÖ –°–æ–∑–¥–∞–Ω–∞ –∑–æ–Ω–∞ —ç–∫—Å–ø—Ä–µ—Å—Å-–¥–æ—Å—Ç–∞–≤–∫–∏ —Ä–∞–¥–∏—É—Å–æ–º ${radiusKm}–∫–º, –≤—Ä–µ–º—è ${deliveryTimeMinutes}–º–∏–Ω`);
      return polygon.polygon_id;
    } catch (error) {
      console.error('–û—à–∏–±–∫–∞ —Å–æ–∑–¥–∞–Ω–∏—è –∑–æ–Ω—ã —ç–∫—Å–ø—Ä–µ—Å—Å-–¥–æ—Å—Ç–∞–≤–∫–∏:', error);
      throw error;
    }
  }
  
  /**
   * –°–æ–∑–¥–∞–Ω–∏–µ –∑–æ–Ω –¥–æ—Å—Ç–∞–≤–∫–∏ –ø–æ —Ä–∞–π–æ–Ω–∞–º –≥–æ—Ä–æ–¥–∞
   */
  async setupCityDistrictDelivery(districts: CityDistrict[]) {
    const results: DistrictResult[] = [];
    
    for (const district of districts) {
      try {
        const polygon = await this.ozonApi.polygon.createDeliveryPolygon({
          coordinates: district.coordinates
        });

          await this.ozonApi.polygon.bindPolygonToDeliveryMethod({
            delivery_method_id: district.deliveryMethodId,
            warehouse_location: district.warehouseLocation,

              polygon_id: polygon.polygon_id,

          });
          
          console.log(`‚úÖ –†–∞–π–æ–Ω ${district.name}: –ø–æ–ª–∏–≥–æ–Ω ${polygon.polygon_id}`);
        }
      } catch (error) {
        results.push({
          district: district.name,
          status: 'failed',

        console.error(`‚ùå –û—à–∏–±–∫–∞ –Ω–∞—Å—Ç—Ä–æ–π–∫–∏ —Ä–∞–π–æ–Ω–∞ ${district.name}:`, error);
      }
      
      // –ó–∞–¥–µ—Ä–∂–∫–∞ –º–µ–∂–¥—É –∑–∞–ø—Ä–æ—Å–∞–º–∏
```
